/*CONFIRM ENDING*/
function confirmending(text,str){
    var a=text.length;
    var b=str.length;
    var ending=text.substring((a-b),a);

    if (str==ending){
        return true;
    }else {return false;}
};

/*REMOVE STRING*/

function removestring(str,first){

return str.replace(first,'' );

};


/* REPLACE STRING */

function replacestring(str,first,second){

};
